class ctcpclient         // TCP
{
private:
  int m_clientfd;        // socket-1>=0socket

  unsigned short m_port; // 
public:
  ctcpclient():m_clientfd(-1) {}

  // truefalse
  bool connect(const string &in_ip,const unsigned short in_port)
  {
    if (m_clientfd!=-1) return false; // socket

    m_ip=in_ip; m_port=in_port;       // IP

    // 1socket
    if ( (m_clientfd = socket(AF_INET,SOCK_STREAM,0))==-1) return false;

    // 2
    struct sockaddr_in servaddr;               // IP
    memset(&servaddr,0,sizeof(servaddr));
    servaddr.sin_family = AF_INET;             // AF_INET
    servaddr.sin_port = htons(m_port);         // 

    struct hostent* h;                         // IP()
    if ((h=gethostbyname(m_ip.c_str()))==nullptr ) // //IP
    {
      ::close(m_clientfd); m_clientfd=-1; return false;
    }
    memcpy(&servaddr.sin_addr,h->h_addr,h->h_length); // IP()

    // 
    if (::connect(m_clientfd,(struct sockaddr *)&servaddr,sizeof(servaddr))==-1)  
    {
      ::close(m_clientfd); m_clientfd=-1; return false;
    }

    return true;
  }

  // truefalse
  bool send(const string &buffer)   // bufferconst char *
  {
    if (m_clientfd==-1) return false; // socket

    if ((::send(m_clientfd,buffer.data(),buffer.size(),0))<=0) return false;

    return true;
  }

  // truefalse      
  bool send(void *buffer,const size_t size)   
  {
    if (m_clientfd==-1) return false; // socket

    if ((::send(m_clientfd,buffer,size,0))<=0) return false;

    return true;
  }

  // truefalse
  // buffer-maxlen-
  bool recv(string &buffer,const size_t maxlen)
  { // string1)2
    buffer.clear();         // 
    buffer.resize(maxlen);  // maxlen
    int readn=::recv(m_clientfd,&buffer[0],buffer.size(),0);  // buffer
    if (readn<=0) { buffer.clear(); return false; }
    buffer.resize(readn);   // buffer

    return true;
  }

  // 
  bool close()
  {
    if (m_clientfd==-1) return false; // socket

    ::close(m_clientfd);
    m_clientfd=-1;
    return true;
  }

  // 
  bool sendfile(const string &filename,const size_t filesize)
  {
    // 
    ifstream fin(filename,ios::binary);
    if (fin.is_open() == false) { cout << "" << filename << "\n";  return false; }

    int  onread=0;        // fin.read()  
    int  totalbytes=0;    //  
    char buffer[4096];       // buffer     

    while (true)
    {
      memset(buffer,0,sizeof(buffer));

      // 40964096
      if (filesize-totalbytes>4096) onread=4096;
      else onread=filesize-totalbytes;

      // 
      fin.read(buffer,onread);

      // 
      if (send(buffer,onread)==false) return false;

      // 
      totalbytes=totalbytes+onread;

      if (totalbytes==filesize) break;
    }

    return true;
  }

 ~ctcpclient(){ close(); }

class ctcpserver         // TCP
{
private:
  int    m_listenfd;        // socket-1
  int    m_clientfd;        // socket-1
  string m_clientip;        // IP
  unsigned short m_port;    // 
public:
  ctcpserver():m_listenfd(-1),m_clientfd(-1) {}

  // socket
  bool initserver(const unsigned short in_port)
  {
    // 1socket 
    if ( (m_listenfd=socket(AF_INET,SOCK_STREAM,0))==-1) return false;

    m_port=in_port;

    // 2IPsocket 
    struct sockaddr_in servaddr;                // IP
    memset(&servaddr,0,sizeof(servaddr));
    servaddr.sin_family=AF_INET;                // AF_INET
    servaddr.sin_port=htons(m_port);            // 
    servaddr.sin_addr.s_addr=htonl(INADDR_ANY); // IPIP

    // IPsocketIP
    if (bind(m_listenfd,(struct sockaddr *)&servaddr,sizeof(servaddr))==-1)
    { 
      close(m_listenfd); m_listenfd=-1; return false; 
    }

    // 3socket
    if (listen(m_listenfd,5) == -1 ) 
    { 
      close(m_listenfd); m_listenfd=-1; return false;
    }

    return true;
  }

  // 
  // accept()
  bool accept()
  {
    struct sockaddr_in caddr;        //   
    socklen_t addrlen=sizeof(caddr); // struct sockaddr_in
    if ((m_clientfd=::accept(m_listenfd,(struct sockaddr *)&caddr,&addrlen))==-1) return false;

    m_clientip=inet_ntoa(caddr.sin_addr);  // 

    return true;
  }

  // IP()
  const string & clientip() const
  {
    return m_clientip;
  }

  // truefalse
  bool send(const string &buffer)   
  {
    if (m_clientfd==-1) return false;

    if ( (::send(m_clientfd,buffer.data(),buffer.size(),0))<=0) return false;

    return true;
  }

  // truefalse
  // buffer-maxlen-
  bool recv(string &buffer,const size_t maxlen)
  { 
    buffer.clear();         // 
    buffer.resize(maxlen);  // maxlen
    int readn=::recv(m_clientfd,&buffer[0],buffer.size(),0);  // buffer
    if (readn<=0) { buffer.clear(); return false; }
    buffer.resize(readn);   // buffer

    return true;
  }

  // truefalse
  // buffer-size-
  bool recv(void *buffer,const size_t size)
  {
    if (::recv(m_clientfd,buffer,size,0)<=0) return false;

    return true;
  }

  // socket
  bool closelisten()
  {
    if (m_listenfd==-1) return false; 

    ::close(m_listenfd);
    m_listenfd=-1;
    return true;
  }

  // socket
  bool closeclient()
  {
    if (m_clientfd==-1) return false;

    ::close(m_clientfd);
    m_clientfd=-1;
    return true;
  }

  // 
  bool recvfile(const string &filename,const size_t filesize)
  {
    ofstream fout;
    fout.open(filename,ios::binary);
    if (fout.is_open() == false) { cout << "" << filename << "\n";  return false; }

    int  totalbytes=0;        // 
    int  onread=0;            // 
    char buffer[4096];           // 

    while (true)
    {
      // 
      if (filesize-totalbytes>4096) onread=4096;
      else onread=filesize-totalbytes;

      // 
      if (recv(buffer,onread)==false) return false;

      // 
      fout.write(buffer,onread);

      // 
      totalbytes=totalbytes+onread;

      if (totalbytes==filesize) break;
    }

    return true;
  }

 ~ctcpserver() { closelisten(); closeclient(); }
};
