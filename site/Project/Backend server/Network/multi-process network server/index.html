class ctcpserver         // TCP
{
private:
  int    m_listenfd;        // socket-1
  int    m_clientfd;        // socket-1
  string m_clientip;        // IP
  unsigned short m_port;    // 
public:
  ctcpserver():m_listenfd(-1),m_clientfd(-1) {}

  // socket
  bool initserver(const unsigned short in_port)
  {
    // 1socket 
    if ( (m_listenfd=socket(AF_INET,SOCK_STREAM,0))==-1) return false;

    m_port=in_port;

    // 2IPsocket 
    struct sockaddr_in servaddr;                // IP
    memset(&servaddr,0,sizeof(servaddr));
    servaddr.sin_family=AF_INET;                // AF_INET
    servaddr.sin_port=htons(m_port);            // 
    servaddr.sin_addr.s_addr=htonl(INADDR_ANY); // IPIP

    // IPsocketIP
    if (bind(m_listenfd,(struct sockaddr *)&servaddr,sizeof(servaddr))==-1)
    { 
      close(m_listenfd); m_listenfd=-1; return false; 
    }

    // 3socket
    if (listen(m_listenfd,5) == -1 ) 
    { 
      close(m_listenfd); m_listenfd=-1; return false;
    }

    return true;
  }

  // 
  // accept()
  bool accept()
  {
    struct sockaddr_in caddr;        //   
    socklen_t addrlen=sizeof(caddr); // struct sockaddr_in
    if ((m_clientfd=::accept(m_listenfd,(struct sockaddr *)&caddr,&addrlen))==-1) return false;

    m_clientip=inet_ntoa(caddr.sin_addr);  // 

    return true;
  }

  // IP()
  const string & clientip() const
  {
    return m_clientip;
  }

  // truefalse
  bool send(const string &buffer)   
  {
    if (m_clientfd==-1) return false;

    if ( (::send(m_clientfd,buffer.data(),buffer.size(),0))<=0) return false;

    return true;
  }

  // truefalse
  // buffer-maxlen-
  bool recv(string &buffer,const size_t maxlen)
  { 
    buffer.clear();         // 
    buffer.resize(maxlen);  // maxlen
    int readn=::recv(m_clientfd,&buffer[0],buffer.size(),0);  // buffer
    if (readn<=0) { buffer.clear(); return false; }
    buffer.resize(readn);   // buffer

    return true;
  }

  // socket
  bool closelisten()
  {
    if (m_listenfd==-1) return false; 

    ::close(m_listenfd);
    m_listenfd=-1;
    return true;
  }

  // socket
  bool closeclient()
  {
    if (m_clientfd==-1) return false;

    ::close(m_clientfd);
    m_clientfd=-1;
    return true;
  }

 ~ctcpserver() { closelisten(); closeclient(); }
};
