template <class TT, int MaxLength>
class squeue
{
private:
  bool m_inited;              // true-false-
  TT   m_data[MaxLength];     // 
  int  m_head;                // 
  int  m_tail;                // 
  int  m_length;              //     
  squeue(const squeue &) = delete;             // 
  squeue &operator=(const squeue &) = delete;  // 
public:

  squeue() { init(); }  // 

  // 
  // 
  void init()  
  { 
    if (m_inited!=true)      // 
    { 
      m_head=0;              // 
      m_tail=MaxLength-1;    // 
      m_length=0;            // 
      memset(m_data,0,sizeof(m_data));  // 
      m_inited=true; 
    }
  }

  // false-true-
  bool push(const TT &ee)
  {
    if (full() == true)
    {
      cout << "\n"; return false;
    }

    // 
    m_tail=(m_tail+1)%MaxLength;  // 
    m_data[m_tail]=ee;
    m_length++;    

    return true;
  }

  // >=0-
  int  size()                   
  {
    return m_length;    
  }

  // true-false-
  bool empty()                    
  {
    if (m_length == 0) return true;    

    return false;
  }

  // true-false-
  bool full()
  {
    if (m_length == MaxLength) return true;    

    return false;
  }

  // 
  TT& front()
  {
    return m_data[m_head];
  }

  // false-true-
  bool pop()
  {
    if (empty() == true) return false;

    m_head=(m_head+1)%MaxLength;  // 
    m_length--;    

    return true;
  }

  // 
  // cout
  void printqueue()                    
  {
    for (int ii = 0; ii < size(); ii++)
    {
      cout << "m_data[" << (m_head+ii)%MaxLength << "],value=" \
           << m_data[(m_head+ii)%MaxLength] << endl;
    }
  }
};

bool csemp::init(key_t key, unsigned short value, short sem_flg) {
  if (m_semid != -1) return false;  // 

  m_sem_flg = sem_flg;

  // semget(key,1,0666|IPC_CREAT)
  // 01
  // 

  // 
  // 1
  // 2
  // 3) 

  // 
  if ((m_semid = semget(key, 1, 0666)) == -1) {
    // 
    if (errno == ENOENT) {
      // IPC_EXCL
      if ((m_semid = semget(key, 1, 0666 | IPC_CREAT | IPC_EXCL)) == -1) {
        if (errno == EEXIST)  // 
        {
          if ((m_semid = semget(key, 1, 0666)) == -1) {
            perror("init 1 semget()");
            return false;
          }
          return true;
        } else  // 
        {
          perror("init 2 semget()");
          return false;
        }
      }

      // value
      union semun sem_union;
      sem_union.val = value;  // 
      if (semctl(m_semid, 0, SETVAL, sem_union) < 0) {
        perror("init semctl()");
        return false;
      }
    } else {
      perror("init 3 semget()");
      return false;
    }
  }

  return true;
}

// Pvalue00
bool csemp::wait(short value) {
  if (m_semid == -1) return false;

  struct sembuf sem_b;
  sem_b.sem_num = 0;     // 0
  sem_b.sem_op = value;  // Pvalue0
  sem_b.sem_flg = m_sem_flg;
  if (semop(m_semid, &sem_b, 1) == -1) {
    perror("p semop()");
    return false;
  }

  return true;
}

// Vvalue
bool csemp::post(short value) {
  if (m_semid == -1) return false;

  struct sembuf sem_b;
  sem_b.sem_num = 0;     // 0
  sem_b.sem_op = value;  // Vvalue0
  sem_b.sem_flg = m_sem_flg;
  if (semop(m_semid, &sem_b, 1) == -1) {
    perror("V semop()");
    return false;
  }

  return true;
}

// -1
int csemp::getvalue() { return semctl(m_semid, 0, GETVAL); }

// 
bool csemp::destroy() {
  if (m_semid == -1) return false;

  if (semctl(m_semid, 0, IPC_RMID) == -1) {
    perror("destroy semctl()");
    return false;
  }

  return true;
}

csemp::~csemp() {}
